1：连接到数据库服务器
mysql -h host -u root -p 密码

2：查看所有库
show databases;

3：选库
use 库名

4：查看库下面的表
show tables

5：建表
create table msg(       //msg 是你要建的表名
id int auto_increment primary key, // 对表中的项目进行定义
content varchar(200),
pubtime int 
)charset utf8

6：告诉服务器你的字符集
set name gbk/utf8/....;

7:添加数据
insert into msg(id,content,pubtime) values (1,'逗你玩‘，12443）；

8：查询所有数据
select * from msg;

9:按id查询一行
select * from msg;

10:快速清空表
truncate 表名
